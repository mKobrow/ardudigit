Lösungscode Modul 2


						#include <Ardudigit.h>

						int buttonPin = 7;
						I2C_LCD lcd(0x27);

						void setup() {
						  pinMode(buttonPin, INPUT_PULLUP);
						  lcd.init();
						  lcd.print("Taste druecken..");
						}

						void loop() {
						  if (digitalRead(buttonPin) == LOW) {
						    lcd.clear();
						    for (int i = 5; i > 0; i--) {
						      lcd.printf("%d ", i);
						      delay(1000);
						    }
						    lcd.clear();
						    lcd.print("Taste druecken..");
						  }
						}
